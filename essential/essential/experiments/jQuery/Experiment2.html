About the Experiment

In this experiment, I experimented with jQuery effects to create different types of alerts and to change text formatting in HTML. Also implemented buttons to remove the text of different types when clicked.

Code Snippet - Javascript

$(document).ready(function () {
    $("#button1").click(function () {
        alert("Text: " + $("#p1").text());
    });
    $("#button2").click(function () {
        alert("HTML: " + $("#p1").html());
    });
    $("#button3").click(function () {
        alert("Value: " + $("#input").val());
    });
    $("#button4").click(function () {
        alert($("#thispage").attr("href"));
    });
    $("#button5").click(function () {
        $("#p2").text("Changed text on click");
    });
    $("#button6").click(function () {
        $("#p2").html(function () {
            return "Hello world";
        });
    });
    $("#button7").click(function () {
        $("#p3").append(" Append text");
    });
    $("#button8").click(function () {
        $("#p3").prepend(" Prepend text");
    });
    $("#button9").click(function () {
        var txt1 = "Created this text using HTML";              // Create text with HTML
        var txt2 = $("").text("Created this text using jQuery");  // Create text with jQuery
        var txt3 = document.createElement("p");
        txt3.innerHTML = "Created this text using DOM";               // Create text with DOM
        $("#app").append(txt1, txt2, txt3);     // Append new elements
    });
    $("#button10").click(function () {
        $("#p4").before("used .before");
    });
    $("#button11").click(function () {
        $("#p4").after(" used .after");
    });
    $("#button12").click(function () {
        $("#p5").remove();
    });
    $("#button13").click(function () {
        $("#p6").remove(".italic");
    });
    $("#button14").click(function () {
        $("#p6").remove(".bold");
    });
});
