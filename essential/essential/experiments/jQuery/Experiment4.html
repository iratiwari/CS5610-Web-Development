About the Experiment

In this experiment, I experimented with jQuery manpiulating dimensions of boxes. The dimensions are shown when each button next to the box is clicked, or resized as required.

Code Snippet - Javascript

        $(document).ready(function () {
    $("#button1").click(function () {
        var txt = "
";
        txt += "Width of div: " + $("#div5").width() + "
";
        txt += "Height of div: " + $("#div5").height();
        $("#div5").html(txt);
    });
    $("#button2").click(function ()
    {
        var txt = "";
        txt += "Width of div: " + $("#div6").width() + "
";
        txt += "Height of div: " + $("#div6").height() + "
";
        txt += "Inner width of div: " + $("#div6").innerWidth() + "
";
        txt += "Inner height of div: " + $("#div6").innerHeight();
        $("#div6").html(txt);
    });
    $("#button3").click(function ()
    {
        var txt = "";
        txt += "Width of div: " + $("#div7").width() + "
";
        txt += "Height of div: " + $("#div7").height() + "
";
        txt += "Outer width: " + $("#div7").outerWidth() + "
";
        txt += "Outer height: " + $("#div7").outerHeight();
        $("#div7").html(txt);
    });
    $("#button4").click(function () {
        $("#div8").width(100).height(100);
    });
});
